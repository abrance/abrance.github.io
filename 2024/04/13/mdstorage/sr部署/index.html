resource check

#!/bin/bash

echo "" > ~/SC_v1.6.x.linux-x86_64/check.sh

cat << EOF > /opt/installer/config/resource_check_config.yaml
osKernelArch:
  type: in
  options:
  - "ubuntu 20.04、5.4、amd64"
  - "ubuntu 20.04、5.15、amd64"  
  - "centos 7、3.10、amd64"
nproc:
  type: range
  minimum: 2
  standard: 4
memory:
  type: range
  minimum: 2
  standard: 4
disk:
  type: range
  minimum: 1
  standard: 10
EOF
systemctl restart installer

#!/bin/bash
